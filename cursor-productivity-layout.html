  simulateGpuNodes(count) {
    const newNodes = [];
    
    for (let i = 0; i < count; i++) {

        gpuUsage: Math.floor(Math.random() * 100),

        bandwidth: Math.floor(Math.random() * 1000) + 500,
        status: 'active',
        label: `GPU ${i + 1}`
      };
      
      this.nodes.push(node);
