            function initScene4() {
                // Criar rede neural

                for (let i = 0; i < nodeCount; i++) {

                    node.style.top = `${Math.random() * 90 + 5}%`;
                    neuralNetwork.appendChild(node);
                    nodes.push(node);
                    
                    // Animar n√≥s aparecendo gradualmente
                    setTimeout(() => {
                        node.style.opacity = '1';
                    }, 1000 + i * 100);
                }
                
                // Criar conex√µes entre n√≥s
                for (let i = 0; i < nodeCount; i++) {
                    for (let j = i + 1; j < nodeCount; j++) {
                        // Apenas criando conex√µes entre alguns n√≥s, n√£o todos
                        if (Math.random() > 0.7) continue;
                        
                        const connection = document.createElement('div');
                        connection.className = 'connection';
                        
                        // Posi√ß√£o e geometria da linha de conex√£o
                        const node1Rect = nodes[i].getBoundingClientRect();
                        const node2Rect = nodes[j].getBoundingClientRect();
                        
                        const x1 = node1Rect.left + node1Rect.width / 2;
                        const y1 = node1Rect.top + node1Rect.height / 2;
                        const x2 = node2Rect.left + node2Rect.width / 2;
                        const y2 = node2Rect.top + node2Rect.height / 2;
                        
                        const distance = Math.sqrt(Math.pow(x2 - x1, 2) + Math.pow(y2 - y1, 2));
                        const angle = Math.atan2(y2 - y1, x2 - x1) * 180 / Math.PI;
                        
                        connection.style.width = `${distance}px`;
                        connection.style.transform = `rotate(${angle}deg)`;
                        connection.style.left = `${x1}px`;
                        connection.style.top = `${y1}px`;
                        
                        neuralNetwork.appendChild(connection);
                        
                        // Animar conex√µes aparecendo gradualmente
                        setTimeout(() => {
                            connection.style.opacity = '0.5';
                        }, 3000 + (i + j) * 50);
                    }
                }
                
                // Animar o cart√£o de Runa
                setTimeout(() => {
                    document.querySelector('.rune-card').style.opacity = '1';
                    
                    // Anima√ß√£o suave de rota√ß√£o do cart√£o
                    let rotationY = 30;
                    let rotationDirection = 1;
                    
                    setInterval(() => {
                        rotationY += 0.1 * rotationDirection;
                        
                        if (rotationY > 40) rotationDirection = -1;
                        if (rotationY < 20) rotationDirection = 1;
                        
                        document.querySelector('.card-inner').style.transform = `rotateY(${rotationY}deg)`;
                    }, 50);
                }, 5000);
                
                // C√≥digo Morse para "Nakamoto"
                // N (-.): longo, curto
                // A (.-): curto, longo
                // K (-.-): longo, curto, longo
                // A (.-): curto, longo
                // M (--): longo, longo
                // O (---): longo, longo, longo
                // T (-): longo
                // O (---): longo, longo, longo
                const morseCode = [
                    {duration: 800, pause: 200}, {duration: 200, pause: 500}, // N
                    {duration: 200, pause: 200}, {duration: 800, pause: 500}, // A
                    {duration: 800, pause: 200}, {duration: 200, pause: 200}, {duration: 800, pause: 500}, // K
                    {duration: 200, pause: 200}, {duration: 800, pause: 500}, // A
                    {duration: 800, pause: 200}, {duration: 800, pause: 500}, // M
                    {duration: 800, pause: 200}, {duration: 800, pause: 200}, {duration: 800, pause: 500}, // O
                    {duration: 800, pause: 500}, // T
                    {duration: 800, pause: 200}, {duration: 800, pause: 200}, {duration: 800, pause: 1000} // O
                ];
                
                // Animar o c√≥digo Morse
                const morseElement = document.getElementById('morse-code');
                
                setTimeout(() => {
                    let currentIndex = 0;
                    
                    function playNextMorse() {
                        if (currentIndex >= morseCode.length) return;
                        
                        const {duration, pause} = morseCode[currentIndex];
                        
                        morseElement.style.opacity = '1';
                        
                        setTimeout(() => {
                            morseElement.style.opacity = '0';
                            
                            setTimeout(() => {
                                currentIndex++;
                                playNextMorse();
                            }, pause);
                        }, duration);
                    }
                    
                    playNextMorse();
                }, 8000);
            }

            function showScene(index) {

                        if (i === 0) initScene1();
                        else if (i === 1) initScene2();

                    }
                });

            // Determinar qual cena mostrar com base no tempo atual
            function updateSceneBasedOnTime() {
                let currentScene = 0;
                
                for (let i = 1; i < sceneStartTimes.length; i++) {
                    if (currentTime >= sceneStartTimes[i]) {
                        currentScene = i;
                    }
                }
                
                showScene(currentScene);
            }
            
            // Atualizar o display de tempo
            function updateTimeDisplay() {
                const minutes = Math.floor(currentTime / 60);
                const seconds = Math.floor(currentTime % 60);
                const totalMinutes = Math.floor(totalDuration / 60);
                const totalSeconds = Math.floor(totalDuration % 60);
                
                timeDisplay.textContent = `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')} / ${totalMinutes.toString().padStart(2, '0')}:${totalSeconds.toString().padStart(2, '0')}`;
            }
            
            // Atualizar progresso do v√≠deo
            function updateProgress() {
                const progress = (currentTime / totalDuration) * 100;
                progressIndicator.style.width = `${progress}%`;
            }
            
            // Fun√ß√£o para animar o "v√≠deo"
            function animate() {
                if (isPlaying) {
                    currentTime += 0.05; // Incremento de tempo
                    
                    if (currentTime >= totalDuration) {
                        currentTime = totalDuration;
                        isPlaying = false;
                        playPauseBtn.textContent = '‚ñ∂';
                    }
                    
                    updateProgress();
                    updateTimeDisplay();
                    updateSceneBasedOnTime();
                    
                    animationFrameId = requestAnimationFrame(animate);
                }
            }
            
            // Event listeners para controles de v√≠deo
            playPauseBtn.addEventListener('click', () => {
                isPlaying = !isPlaying;
                
                if (isPlaying) {
                    playPauseBtn.textContent = '‚ùö‚ùö';
                    animationFrameId = requestAnimationFrame(animate);
                } else {
                    playPauseBtn.textContent = '‚ñ∂';
                    cancelAnimationFrame(animationFrameId);
                }
            });
            
            // Clique na barra de progresso
            progressBar.addEventListener('click', (event) => {
                const rect = progressBar.getBoundingClientRect();
                const clickPosition = (event.clientX - rect.left) / rect.width;
                
                currentTime = clickPosition * totalDuration;
                
                updateProgress();
                updateTimeDisplay();
                updateSceneBasedOnTime();
            });
            
            // Inicializar a primeira cena
            showScene(0);
            updateTimeDisplay();
