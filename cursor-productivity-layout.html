class GpuMeshNetworkVisualizer {
  constructor(options = {}) {
    this.container = options.container || '#network-container';
    this.nodes = [];
    this.edges = [];
    this.renderer = null;
    this.gpuClient = options.gpuClient;
    
    this.setupEventListeners();
    this.setupGPUClientListeners();
    
    // Adicionado: Event listener para novos nós GPU
    window.addEventListener("gpu-mesh:new-node", (e) => {
      const newNode = e.detail;
      this.nodes.push(newNode);
      this.render();
    });
  }

  simulateGpuNodes(count) {
    const newNodes = [];
    
    for (let i = 0; i < count; i++) {
      const id = `node-${Math.random().toString(36).substr(2, 6)}`;
      const node = {
        id,
        type: 'gpu',
        gpuUsage: Math.floor(Math.random() * 100),
        latency: Math.floor(Math.random() * 200),
        bandwidth: Math.floor(Math.random() * 1000) + 500,
        status: 'active',
        label: `GPU ${i + 1}`
      };
      
      this.nodes.push(node);
      newNodes.push(node);
      
      console.log(`[GPU Mesh] Novo nó adicionado: ${id}`);
    }
    
    // Importação do módulo de notificação
    import('./mesh-notify.js').then(module => {
      const { notifyOwl } = module;
      notifyOwl("Adição de novos nós GPU", { totalAdicionados: count });
    });
    
    this.render();
    return newNodes;
  }
}
